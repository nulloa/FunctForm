model{
  
  for (i in 1:n) {
    y[i] ~ dbinom(theta[i], num[i])
    logit(theta[i]) <- ltheta[i]
    ltheta[i] ~ dnorm(mu[group[i]], sigma[group[i]])
  }
  
  for (g in 1:nG) {
    mu[g]    ~ dnorm(m_x, t_m)
    sigma[g] ~ dt(0, t_s, 1) T(0,)
  }
  
  m_x ~ dnorm(mx, 1/vmx)

  t_m <- 1/sqrt(tau_m)
  tau_m ~ dt(0, 1/vm, 1) T(0,)
  
  t_s <- 1/sqrt(tau_s)
  tau_s ~ dt(0, 1/vs, 1) T(0,)

  }
