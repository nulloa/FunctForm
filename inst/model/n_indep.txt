model{
  
  for (i in 1:n) {
    y[i] ~ dbinom(theta[i], num[i])
    logit(theta[i]) <- ltheta[i]
    ltheta[i] = (1/(sqrt(2*3.14)*sigma[group[i]]^2))*exp(-(x - mu[group[i]])^2 / (2*sigma[group[i]]^2))
  }
  
  for (g in 1:nG) {
    mu[g] ~ dnorm(mx, 1/vm)
    sigma[g] <- 1/(sig[g])^2
    sig[g] ~ dt(0, 1/vs, 1) T(0,)
  }
  
  }
