model{
  
  for (i in 1:n) {
    y[i] ~ dbinom(theta[i], num[i])
    logit(theta[i]) <- ltheta[i]
    ltheta[i] ~ dlnorm(mu[group[i]], sigma[group[i]])
  }
  
  for (g in 1:nG) {
    mu[g]    ~ dnorm(m_x, t_m)
    1/sqrt(sigma[g]) <- sig[g]
    sig[g] ~ dt(0, t_s, 4) T(0,)
  }
  
  m_x ~ dnorm(mx, 1/vmx)

  1/sqrt(t_m) <- tau_m
  tau_m ~ dt(0, 1/vtm, 1) T(0,)
  
  1/sqrt(tau_s) <- tau_s
  tau_s ~ dt(0, 1/vts, 1) T(0,)

  }
