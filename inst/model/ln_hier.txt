model{
  
  for (i in 1:n) {
    y[i] ~ dbinom(theta[i], num[i])
    logit(theta[i]) <- ltheta[i]
    ltheta[i] = (1/(sqrt(2*3.14)*sigma[group[i]]*x))*exp(-(log(x) - mu[group[i]])^2 / (2*sigma[group[i]]^2))
  }
  
  for (g in 1:nG) {
    mu[g]    ~ dnorm(m_x, t_m)
    sigma[g] <- 1/(sig[g])^2
    sig[g] ~ dt(0, t_s, 4) T(0,)
  }
  
  m_x ~ dnorm(mx, 1/vmx)

  t_m <- 1/(tau_m)^2
  tau_m ~ dt(0, 1/vtm, 1) T(0,)
  
  t_s <- 1/(tau_s)^2
  tau_s ~ dt(0, 1/vts, 1) T(0,)

  }
